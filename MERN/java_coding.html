Java
import java.util.Scanner;

public class PrimeNumber {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        System.out.print("Enter a number: ");
        int number = scanner.nextInt();

        if (isPrime(number)) {
            System.out.println(number + " is a prime number.");
        } else {
            System.out.println(number + " is not a prime number.");
        }
    }

    public static boolean isPrime(int num) {
        if (num <= 1) return false;
        for (int i = 2; i <= Math.sqrt(num); i++) {
            if (num % i == 0) return false;
        }
        return true;
    }
}

public class PalindromeNumber {
    public static void main(String[] args) {
        int number = 1001;

            System.out.println(number + " is a palindrome.");
        else
            System.out.println(number + " is not a palindrome.");
    }

    public static boolean isPalindrome(int num) {
        int original = num, reversed = 0;
        while (num != 0) {
            int digit = num % 10;
            reversed = reversed * 10 + digit;
            num = num / 10;
        }
        return original == reversed;
    }
}

        System.out.println("Sum of digits of " + number + " is: " + sum);
    }

        int sum = 0;

public class Anagram {
    public static void main(String[] args) {
        System.out.println(areAnagrams("listen", "silent")); // true
    }

    static boolean areAnagrams(String s1, String s2) {
        if (s1.length() != s2.length()) return false;

        int[] charCount = new int[256];
        for (int i = 0; i < s1.length(); i++) {
            charCount[s1.charAt(i)]++;
            charCount[s2.charAt(i)]--;
        }

        for (int count : charCount)
            if (count != 0) return false;

        return true;
    }
}

public class RunLengthEncoding {
    public static void main(String[] args) {
        System.out.println(encode("aabbcccdd"));
    }

    public static String encode(String str) {
        StringBuilder result = new StringBuilder();
        int count = 1;

        for (int i = 0; i < str.length(); i++) {
            if (i+1 < str.length() && str.charAt(i) == str.charAt(i+1))
                count++;
            else {
                result.append(str.charAt(i)).append(count);
                count = 1;
            }
        }
        return result.toString();
    }
}

#13Hard
Longest Substring Without Repeating Characters
Java
import java.util.HashSet;

public class LongestSubstring {
    public static void main(String[] args) {
        System.out.println(lengthOfLongest("abcabcbb")); // 3
        System.out.println(lengthOfLongest("bbbbb"));    // 1
        System.out.println(lengthOfLongest("pwwkew"));   // 3
    }

    public static int lengthOfLongest(String s) {
        HashSet<Character> set = new HashSet<>();
        int maxLen = 0, start = 0, end = 0;

        while (end < s.length()) {
            char cur = s.charAt(end);
            if (!set.contains(cur)) {
                set.add(cur);
                maxLen = Math.max(maxLen, end - start + 1);
                end++;
            } else {
                set.remove(s.charAt(start));
                start++;
            }
        }
        return maxLen;
    }
}
Output
"abcabcbb"→3, "bbbbb"→1, "pwwkew"→3

Sort Array Without Built-in Methods (Selection Sort)
Java
public class SelectionSort {
    public static void main(String[] args) {
        int[] array = {5, 2, 9, 1, 6};
        selectionSort(array);

        System.out.print("Sorted: ");
        for (int n : array) System.out.print(n + " ");
    }

    public static void selectionSort(int[] arr) {
        for (int i = 0; i < arr.length-1; i++) {
            int minIdx = i;
            for (int j = i+1; j < arr.length; j++)
                if (arr[j] < arr[minIdx]) minIdx = j;

            int temp = arr[i];
            arr[i] = arr[minIdx];
            arr[minIdx] = temp;
        }
    }
}

        System.out.println("Missing: " + findMissing(array));
    }

    public static int findMissing(int[] arr) {
        int n = arr.length + 1;

#01Hard
Implement hashCode() and equals()
Java
import java.util.Objects;

public class Student {
    private int id;
    private String name;

    public Student(int id, String name) {
        this.id = id; this.name = name;
    }

    @Override
    public int hashCode() {
        return Objects.hash(id, name);
    }

    @Override
    public boolean equals(Object obj) {
        if (this == obj) return true;
        if (obj == null || getClass() != obj.getClass())
            return false;
        Student s = (Student) obj;
        return id == s.id && Objects.equals(name, s.name);
    }

    public static void main(String[] args) {
        Student s1 = new Student(1, "Alice");
        Student s2 = new Student(2, "Bob");
        Student s3 = new Student(1, "Alice");

        System.out.println(s1.equals(s2)); // false
        System.out.println(s1.equals(s3)); // true
        System.out.println(s1.hashCode() == s3.hashCode()); // true
    }
}
Output
s1.equals(s2)=false | s1.equals(s3)=true | hashCodes match for equal objects

40 Most Asked Java Interview Questions

Q01
What is Java?
Java is a high-level, object-oriented programming language developed by Sun Microsystems that enables development of platform-independent applications through bytecode executed on the JVM.
Q02
What are the 4 pillars of OOP in Java?
Encapsulation — wrapping data and methods together. Inheritance — child class acquiring parent properties. Polymorphism — many forms via overloading/overriding. Abstraction — hiding complex implementation details.
Q03
Difference between JDK, JRE, and JVM?
JDK = Development tools + JRE (for developers). JRE = Runtime libraries + JVM (for running apps). JVM = Java interpreter + JIT + Garbage Collector (executes bytecode).
Q04
What is bytecode in Java?
When javac compiles a class, it generates a .class file containing bytecode — a machine-independent set of instructions executed by the JVM on any platform.
Q05
Why is Java platform independent?
Java compiles source code into .class files containing bytecode. The JVM — available on all major platforms — interprets this bytecode. Code compiled on Windows runs on Linux or any platform with a JVM.
Q06
Difference between C++ and Java?
Java is platform-independent; C++ is platform-dependent. Java has no pointers or operator overloading. Java supports automatic garbage collection and multithreading natively; C++ does not.
Q07
What is JIT compiler?
The Just-In-Time compiler is part of the JRE that compiles bytecode to native machine code at runtime. It improves Java app performance by caching compiled code instead of interpreting each time.
Q08
Difference between this() and super()?
this() calls another constructor within the same class. super() calls the superclass constructor. Both must be the first statement if present in a constructor.
Q09
What is multithreading in Java?
Multithreading divides a Java program into multiple threads that run in parallel, sharing memory space. It improves application performance by enabling concurrent task execution on multi-core processors.
Q10
What is a ClassLoader?
A ClassLoader dynamically loads, links, and initializes classes at runtime. Java uses three types: Bootstrap (core Java classes), Extension (ext directory), and System/Application (classpath classes).
Q11
What is a Java enum?
An enum (enumeration) is a special data type representing named constant values. Introduced in Java 5, it provides type-safe constants declared with the enum keyword: enum Day { MON, TUE, WED }
Q12
Difference between equals() and == in Java?
equals() is a method that compares object content/values. == is an operator that compares references (memory addresses) for objects and values for primitives.
Q13
Lambda expressions and functional interfaces?
A functional interface has exactly one abstract method. Lambda expressions are anonymous implementations of functional interfaces, providing a concise syntax. They are interrelated — lambdas are expressions that implement functional interfaces.
Q14
Roles of final, finally, and finalize?
final (keyword): prevents class inheritance, method overriding, variable reassignment. finally (block): always executes after try-catch. finalize (method): called by GC before object is garbage collected.
Q15
Difference between a class and an object?
A class is a blueprint or template defining properties and behavior. An object is a runtime instance of a class representing a real-world entity. Classes are defined once; objects are created many times.
Q16
What is inheritance in Java?
Inheritance allows a child class to acquire properties and methods of a parent class using the extends keyword. It promotes code reuse. Java supports single, multilevel, and hierarchical inheritance (but NOT multiple with classes).
Q17
What is polymorphism?
Polymorphism means "many forms" — a single method behaves differently based on context. Compile-time: method overloading (same name, different params). Runtime: method overriding (subclass redefines parent method).
Q18
What is encapsulation?
Encapsulation bundles data (fields) and methods into a single class while restricting direct access using private. External access is controlled through public getters and setters, ensuring data integrity.
Q19
What is abstraction?
Abstraction hides complex implementation details while exposing only essential features. Achieved via abstract classes (partial abstraction) and interfaces (full abstraction in older Java; default methods added in Java 8).
Q20
Abstract class vs Interface?
Abstract class uses extends, can have implemented methods, constructors, fields. Interface uses implements, allows multiple implementation, Java 8+ supports default/static methods. You can extend only one abstract class but implement multiple interfaces.
Q21
Can Java support multiple inheritance?
Java does NOT support multiple inheritance through classes (to avoid the Diamond Problem). However, it achieves multiple inheritance through interfaces — a class can implement multiple interfaces.
Q22
Method overloading vs method overriding?
Overloading: same method name, different parameter types/count in the same class (compile-time). Overriding: subclass provides its own implementation of a parent method with the same signature (runtime).
Q23
Can we write code after a throw statement?
No. After a throw statement, the JVM stops execution. Any code written after it causes a compile-time error: "unreachable code."
Q24
Can we nest try statements in Java?
Yes. Java supports nested try statements — you can place a try block inside another try block. Each has its own catch block to handle exceptions at different levels.
Q25
What is a constructor?
A constructor is a special method with the same name as the class and no return type. It initializes a new object. Types: Default (no-arg), Parameterized (with args), and Copy constructor.
Q26
Default value of local variables?
Local variables in Java are NOT initialized to any default value. Neither primitives nor object references have defaults. Using an uninitialized local variable causes a compile-time error.
Q27
What is an object in Java?
An object is a runtime instance of a class that encapsulates state (fields) and behavior (methods). It represents a real-world entity and is the fundamental building block of object-oriented programming.
Q28
Advantages of Packages in Java?
Packages avoid name clashes between classes. They provide easier access control. They allow hidden classes not visible outside the package. They make it easier to locate related classes and organize large projects.
Q29
Can a constructor be final?
No. Constructors cannot be declared final. The final keyword applies to classes (prevent inheritance), methods (prevent overriding), and variables (prevent reassignment) — not constructors.
Q30
What happens if constructor is protected?
A protected constructor is accessible only within the same package or by subclasses (even in different packages). It restricts object creation from unrelated external classes — useful in the Factory pattern.
Q31
What is a static method?
A static method belongs to the class, not an instance. It can be called without creating an object (ClassName.method()). It can only directly access other static members (fields/methods).
Q32
Why is the main method static?
The JVM calls main() without creating an instance of the class first. Making it static avoids the need to instantiate the class, preventing extra memory allocation before the program even starts.
Q33
What is a static block?
A static block is executed exactly once when the class is first loaded into memory — before the main() method. It's used for static initialization like loading drivers or setting configuration.
Q34
Which class is the superclass for all classes?
The Object class (java.lang.Object) is the superclass of all classes in Java. Every class implicitly extends Object, which provides methods like equals(), hashCode(), toString(), and getClass().
Q35
System.out vs System.err vs System.in?
System.out: Standard output stream for normal program results. System.err: Standard error stream for exceptions and error messages. System.in: Standard input stream for reading user input (typically via Scanner).
Q36
What is the difference between String, StringBuilder, and StringBuffer?
String is immutable. StringBuilder is mutable and not thread-safe (faster, single-threaded use). StringBuffer is mutable and thread-safe (synchronized, slower). Prefer StringBuilder for string manipulation in loops.
Q37
What is the Collections Framework?
Java Collections Framework provides unified architecture for storing and manipulating groups of objects. Key interfaces: List (ArrayList, LinkedList), Set (HashSet, TreeSet), Map (HashMap, TreeMap), Queue.
Q38
ArrayList vs LinkedList?
ArrayList uses dynamic array — O(1) random access, O(n) insert/delete in middle. LinkedList uses doubly-linked list — O(n) access, O(1) insert/delete at ends. Choose based on access vs modification pattern.
Q39
HashMap vs HashTable?
HashMap is not synchronized (not thread-safe), allows one null key and multiple null values, faster. Hashtable is synchronized (thread-safe), does not allow null keys or values, slower. Use ConcurrentHashMap for thread-safe modern code.
Q40
What is Exception Handling in Java?
Exception handling manages runtime errors using try, catch, finally, throw, throws. Checked exceptions (must handle: IOException) vs Unchecked (runtime: NullPointerException). Prevents abnormal program termination.
