Core JavaScript Concepts
Q1 null vs undefined — What gets logged? Easy ▼ 
// var1 is declared but not assigned
let var1;
console.log(var1);          // undefined
console.log(typeof var1);   // "undefined"

// var2 is explicitly set to null
let var2 = null;
console.log(var2);          // null
console.log(typeof var2);   // "object" ← famous JS bug!
undefined = variable declared but no value assigned yet.
null = intentional absence of a value (explicitly set).
typeof null === "object" is a historic JavaScript bug kept for backward compatibility.
Q2 Hoisting — 3 questions explained Medium ▼ 
// Q1: ReferenceError — foo is not declared at all
console.log(foo); // ❌ ReferenceError
foo = 1;

// Q2: undefined — var is HOISTED (declaration only, not value)
console.log(foo); // undefined
var foo = 2;
// JS engine sees this as:
// var foo; → hoisted to top
// console.log(foo); → undefined
// foo = 2;

// Q3: 3 — assignment happens before log
var foo;
foo = 3;
console.log(foo); // 3
Hoisting: JavaScript moves var declarations (not initializations) to the top of their scope. let/const are hoisted but not initialized (Temporal Dead Zone).

Q8 Shuffle function Medium ▼ 
// Fisher-Yates shuffle — the correct algorithm
const shuffle = arr => {
  const result = [...arr]; // copy to avoid mutation
  for (let i = result.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [result[i], result[j]] = [result[j], result[i]]; // swap
  }
  return result;
};

console.log(shuffle([1,2,3,4,5])); // e.g. [3,1,5,2,4]

// Simple but slightly biased (avoid in production):
const naiveShuffle = arr => [...arr].sort(() => Math.random() - 0.5);
Fisher-Yates is O(n) and produces a truly uniform distribution. The sort()-based method is biased.

// Generic curry — works with any arity function
const curry = fn => {
  return function curried(...args) {
    if (args.length >= fn.length) {
      return fn.apply(this, args); // enough args → call

    return (...moreArgs) => curried(...args, ...moreArgs);

};

Q1 What does `this` log in 4 scenarios? Medium ▼ 
// Q1: global function — this = window (browser) | global (Node) | undefined (strict)
function getItem() { console.log(this); }
getItem(); // window / global / undefined

// Q2: method call — this = the calling object
const item = {
  title: 'Ball',
  getItem() { console.log(this); }
};
item.getItem(); // { title: 'Ball', getItem: ƒ }

// Q3: class method — this = the instance
class Item {
  title = 'Ball';
  getItem() { console.log(this); }
}
new Item().getItem(); // Item { title: 'Ball' }

// Q4: regular function inside class method — this = undefined (strict class body)
class Item2 {
  title = 'Ball';
  getItem() {
    [1,2,3].map(function(item) {
      console.log(this); // undefined ← classic gotcha!
    });
  }
}
// Fix: use arrow function (inherits outer this)
getItem() { [1,2,3].map(item => console.log(this)); } // Item instance
Rule: Arrow functions don't have their own this — they inherit from the enclosing lexical scope. Regular functions get their this from how they are called.

Q3 Debounce function Hard ▼ 
// Debounce: only fires AFTER user stops calling it
const debounce = (fn, delay) => {
  let timerId;
  return (...args) => {
    clearTimeout(timerId);            // reset every call
    timerId = setTimeout(() => {
      fn.apply(this, args);           // call after silence
    }, delay);
  };
};

const saveInput = name => console.log('saveInput', name);
const processChange = debounce(saveInput, 2000);

// Called 5 times rapidly — only fires once after 2s silence
processChange('foo'); processChange('foo'); processChange('foo');

// With immediate option (fires on leading edge too):
const debounceAdvanced = (fn, delay, immediate = false) => {
  let timerId;
  return (...args) => {
    const callNow = immediate && !timerId;
    clearTimeout(timerId);
    timerId = setTimeout(() => {
      timerId = null;
      if (!immediate) fn.apply(this, args);
    }, delay);
    if (callNow) fn.apply(this, args);
  };
};
Use case: Search input autocomplete — don't hit the API on every keystroke, only after the user pauses typing.
Q4 Throttle function Hard ▼ 
// Throttle: fires at most once per interval
const throttle = (fn, interval) => {
  let lastTime = 0;
  return (...args) => {
    const now = Date.now();
    if (now - lastTime >= interval) {
      lastTime = now;
      fn.apply(this, args);
    }
  };
};

// Alternative: setTimeout-based (queues trailing call)
const throttle2 = (fn, interval) => {
  let waiting = false;
  return (...args) => {
    if (!waiting) {
      fn.apply(this, args);
      waiting = true;
      setTimeout(() => waiting = false, interval);
    }
  };
};
Debounce vs Throttle: Debounce = "wait for quiet then fire once." Throttle = "fire at most once per interval regardless."
Q5 Request manager with retry & delay Hard ▼ 
const requestManager = (url, options, attempts) => {
  return new Promise(async (resolve, reject) => {
    for (let i = 0; i < attempts; i++) {
      try {
        const res = await fetch(url, options);
        if (!res.ok) throw new Error(`HTTP ${res.status}`);
        return resolve(await res.json());
      } catch (err) {
        if (i === attempts - 1) return reject(err);
        const delay = 1000 * Math.pow(2, i); // exponential backoff
        console.log(`Retry ${i+1} after ${delay}ms...`);
        await new Promise(r => setTimeout(r, delay));
      }
    }
  });
};

// Usage:
requestManager('http://foo.com', {}, 3)
  .then(response => console.log(response))
  .catch(err => console.log('All attempts failed'));
Exponential backoff: wait 1s → 2s → 4s between retries. This is the real-world production pattern.

Q1 Shallow vs Deep Comparison Hard ▼ 
// Shallow comparison — one level deep
const shallowEqual = (a, b) => {
  if (a === b) return true;
  if (typeof a !== 'object' || a === null) return false;
  const keysA = Object.keys(a);
  const keysB = Object.keys(b);
  if (keysA.length !== keysB.length) return false;
  return keysA.every(key => a[key] === b[key]);
};

// Deep comparison — recursive, handles nested objects
const deepEqual = (a, b) => {
  if (a === b) return true;
  if (typeof a !== typeof b) return false;
  if (typeof a !== 'object' || a === null) return false;

  if (Array.isArray(a) !== Array.isArray(b)) return false;

  const keysA = Object.keys(a);
  const keysB = Object.keys(b);
  if (keysA.length !== keysB.length) return false;

  return keysA.every(key => deepEqual(a[key], b[key]));
};

console.log(deepEqual({a: {b: 1}}, {a: {b: 1}})); // true
console.log(deepEqual({a: {b: 1}}, {a: {b: 2}})); // false
true / false

// Generic memoize — caches results by arguments
const memoize = fn => {

Q5 12-hour to 24-hour time conversion Medium ▼ 
const convertTo24 = time12 => {
  const [time, modifier] = time12.split(' ');
  let [hours, minutes] = time.split(':').map(Number);

  if (modifier === 'AM') {
    if (hours === 12) hours = 0;          // 12:xx AM → 00:xx
  } else {
    if (hours !== 12) hours += 12;        // 1:xx PM → 13:xx
  }

  return `${String(hours).padStart(2,'0')}:${String(minutes).padStart(2,'0')}`;
};

console.log(convertTo24('12:00 AM')); // '00:00'
console.log(convertTo24('12:00 PM')); // '12:00'
console.log(convertTo24('01:30 PM')); // '13:30'
console.log(convertTo24('11:59 PM')); // '23:59'
'00:00' | '12:00' | '13:30' | '23:59'

// O(n) — Map for O(1) parent lookup
const flatToNested = flatList => {
  const map = new Map();
  const roots = [];

  // Pass 1: create nodes with empty children
  flatList.forEach(item => {
    map.set(item.id, {...item, children: []});
  });

  // Pass 2: connect parents to children
  flatList.forEach(({id, parentId}) => {
    if (parentId === null) {
      roots.push(map.get(id));
    } else {
      map.get(parentId).children.push(map.get(id));
    }
  });

  return roots;
};
